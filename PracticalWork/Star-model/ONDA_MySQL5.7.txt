CREATE TABLE f_viagem (
	idviagem			 int,
	receita_taxas_eur		 numeric(12,2),
	num_contentores		 int,
	peso_total_contentores_kg	 numeric(12,3),
	teu_total			 int,
	duracao_viagem_horas	 numeric(10,2),
	estado_mar			 int,
	distancia_km		 numeric(10,1),
	d_tempo_idtempo		 int NOT NULL,
	d_localizacao_idlocalizacao int NOT NULL,
	d_tempo_idtempo1		 int NOT NULL,
	d_condutor_idcondutor	 int NOT NULL,
	d_barco_idbarco		 int NOT NULL,
	PRIMARY KEY(idviagem)
);

CREATE TABLE d_barco (
	idbarco	 int,
	nome		 varchar(512),
	tipo		 varchar(512),
	tamanho	 numeric(10,2),
	pais		 varchar(512),
	capacidadeteu int,
	nomeempresa	 varchar(512),
	PRIMARY KEY(idbarco)
);

CREATE TABLE d_condutor (
	idcondutor	 int,
	nome	 varchar(512),
	sexo	 char(255),
	idade	 int,
	certificacao varchar(512),
	PRIMARY KEY(idcondutor)
);

CREATE TABLE d_localizacao (
	idlocalizacao int,
	cidade	 varchar(512),
	pais		 varchar(512),
	PRIMARY KEY(idlocalizacao)
);

CREATE TABLE d_tempo (
	idtempo	 int,
	datacompleta date,
	dia		 int,
	mes		 int,
	ano		 int,
	trimestre	 int,
	semestre	 int,
	PRIMARY KEY(idtempo)
);
